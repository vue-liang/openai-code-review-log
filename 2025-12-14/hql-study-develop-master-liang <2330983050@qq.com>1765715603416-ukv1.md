public static void quickSort2(int[] arr, int begin, int end) {
    if (begin >= end) {
        return;
    }
    int start = begin;
    int key = arr[end];
    for (int i = begin; i < end; i++) {
        if (arr[i] < key) {
            swap(arr, start, i);
            start++;
        }
    }
    swap(arr, start, end);
    quickSort2(arr, begin, start - 1);
    quickSort2(arr, start + 1, end);
}
